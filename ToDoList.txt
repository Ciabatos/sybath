+Trzeba przerobic do osbonego folderu wszystkie useFetch 
 potem logikę utrzymać w hookach i przypisać do nich fetchowanie
 tu przykład:
 // filepath: methods/hooks/usePlayerAbilities.tsx
import { useFetchPlayerAbilities } from "@/methods/hooks/useFetchPlayerAbilities"
import { useAtomValue } from "jotai"
import { playerAbilitiesAtom } from "@/store/atoms"

export function usePlayerAbilities() {
  useFetchPlayerAbilities()
  const playerAbilities = useAtomValue(playerAbilitiesAtom)

  const getAbilityById = (id: number) => {
    return playerAbilities.find((ability) => ability.id === id)
  }

  return { playerAbilities, getAbilityById }
}
